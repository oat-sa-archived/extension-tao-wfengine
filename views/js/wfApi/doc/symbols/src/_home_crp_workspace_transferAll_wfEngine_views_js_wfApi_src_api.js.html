 56  // STATES  //
 57 /////////////
 58 
 59 if(typeof(finish) != 'function'){
 60 
 61 	//get the highest context 
 62 	var _stateContext = window.top || window;
 63 	
 64 	/**
 65 	 * Define the item's state as finished.
 66 	 * This state can have some consequences.
 67 	 * 
 68 	 * @function
 69 	 * @namespace wfApi
 70 	 */
 71 	function finish(){
 72 		$(_stateContext).trigger(STATE.ITEM.PRE_FINISHED);
 73 		$(_stateContext).trigger(STATE.ITEM.FINISHED);
 74 		$(_stateContext).trigger(STATE.ITEM.POST_FINISHED);
 75 	}
 76 	
 77 	/**
 78 	 * Add a callback that will be executed on finish state.
 79 	 * 
 80 	 * @function
 81 	 * @namespace wfApi
 82 	 * @param {function} callback
 83 	 */
 84 	function onFinish(callback){
 85 		$(_stateContext).bind(STATE.ITEM.FINISHED, callback);
 86 	}
 87 	
 88 	/**
 89 	 * Add a callback that will be executed on finish but before the other callbacks  
 90 	 * 
 91 	 * @function
 92 	 * @namespace wfApi
 93 	 * @param {function} callback
 94 	 */
 95 	function beforeFinish(callback){
 96 		$(_stateContext).bind(STATE.ITEM.PRE_FINISHED, callback);
 97 	}
 98 	
 99 	/**
100 	 * Add a callback that will be executed on finish but after the other callbacks  
101 	 * 
102 	 * @function
103 	 * @namespace wfApi
104 	 * @param {function} callback
105 	 */
106 	function afterFinish(callback){
107 		$(_stateContext).bind(STATE.ITEM.POST_FINISHED, callback);
108 	}
109 
110 }
111 
112 /*
113  * By default, the variables are pushed when the item is finished
114  */
115 afterFinish(forward);

122 /**
123  * instantiate the RecoveryContext
124  * 
125  * @function
126  * @namespace wfApi
127  * @type RecoveryContext
128  */
129 var recoveryCtx = new RecoveryContext();
130 
131 /**
132  * Initialize the interfaces communication for the context recovery.
133  * The source service defines where and how we retrieve contexts.
134  * The destination service defines where and how we send/save the contexts.
135  * 
136  * @function
137  * @namespace wfApi
138  * 
139  * @param {Object} source 
140  * @param {String} [source.type = "sync"] the type of source <b>(sync|async|manual)</b>
141  * @param {Object} [source.data] For the <i>manual</i> source type, set direclty the context data structure
142  * @param {String} [source.url = "/wfEngine/Context/retrieve"] For the <i>sync</i> and <i>async</i> source type, the URL of the remote service
143  * @param {Object} [source.params] the parameters to send to the remote service
144  * @param {String} [source.format = "json"] the data format. <i>Only json is supported in the current version</i> 
145  * @param {String} [source.method = "post"] HTTP method of the sync service <b>(get|post)</b>
146  * 
147  * @param {Object} destination
148  * @param {String} [destination.type = "async"] the type of source <b>(sync|async)</b>
149  * @param {String} [destination.url = "/wfEngine/Context/save"] the URL of the remote service
150  * @param {Object} [destination.params] the common parameters to send to the service
151  * @param {String} [destination.format = "json"] the data format. <i>Only json is supported in the current version</i> 
152  * @param {String} [destination.method = "post"] HTTP method of the service <b>(get|post)</b>
153  */
154 function initRecoveryContext(source, destination){
155 	recoveryCtx.initSourceService(source);
156 	recoveryCtx.initDestinationService(destination);
157 }
158 
159 /**
160  * Retrieve the context identified by the key 
161  * 
162  * @function
163  * @namespace wfApi
164  * 
165  * @param {String} key
166  * @return {Object}
167  */
168 function getRecoveryContext(key){
169 	return recoveryCtx.getContext(key);
170 }
171 
172 /**
173  * Set, send and save a context that could be recovered 
174  * 
175  * @function
176  * @namespace wfApi
177  * 
178  * @param {String} key
179  * @param {Object} data
180  */
181 function setRecoveryContext(key, data){
182 	recoveryCtx.setContext(key, data);
183 	recoveryCtx.saveContext();
184 }
185 
186 /**
187  * Remove a context (once you don't need you to recover it anymore) 
188  * 
189  * @function
190  * @namespace wfApi
191  * 
192  * @param {String} key
193  */
194 function deleteRecoveryContext(key){
195 	setRecoveryContext(key, null);
196 }
