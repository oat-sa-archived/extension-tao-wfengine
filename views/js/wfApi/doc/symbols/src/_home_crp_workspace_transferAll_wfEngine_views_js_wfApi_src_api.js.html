 23 /**
 24  * Trigger the forward control:
 25  * go to the next activity
 26  * 
 27  * @function
 28  * @namespace wfApi
 29  */
 30 function forward(){
 31 	var next = wfApi.context.getElementById('next');
 32 	if(next){
 33 		$(next).trigger('click');
 34 	}
 35 }
 36 
 37 /**
 38  * Trigger the backward control:
 39  * load to the previous activity
 40  * 
 41  * @function
 42  * @namespace wfApi
 43  */
 44 function backward(){
 45 	var back = wfApi.context.getElementById('back');
 46 	if(back){
 47 		$(back).trigger('click');
 48 	}
 49 }
 50 
 51 /**
 52  * Trigger the pause control:
 53  * suspend the current activity and go back to the process control panel
 54  * 
 55  * @function
 56  * @namespace wfApi
 57  */
 58 function pause(){
 59 	var pause = wfApi.context.getElementById('pause');
 60 	if(pause){
 61 		$(pause).trigger('click');
 62 	}
 63 }

 69 
 70 if(typeof(finish) != 'function'){
 71 
 72 	/**
 73 	 * Define the item's state as finished.
 74 	 * This state can have some consequences.
 75 	 * 
 76 	 * @function
 77 	 * @namespace wfApi
 78 	 */
 79 	function finish(){
 80 		$(window).trigger(wf_STATE.ITEM.PRE_FINISHED);
 81 		$(window).trigger(wf_STATE.ITEM.FINISHED);
 82 		$(window).trigger(wf_STATE.ITEM.POST_FINISHED);
 83 	}
 84 	
 85 	/**
 86 	 * Add a callback that will be executed on finish state.
 87 	 * 
 88 	 * @function
 89 	 * @namespace wfApi
 90 	 * @param {function} callback
 91 	 */
 92 	function onFinish(callback){
 93 		$(window).bind(wf_STATE.ITEM.FINISHED, callback);
 94 	}
 95 	
 96 	/**
 97 	 * Add a callback that will be executed on finish but before the other callbacks  
 98 	 * 
 99 	 * @function
100 	 * @namespace wfApi
101 	 * @param {function} callback
102 	 */
103 	function beforeFinish(callback){
104 		$(window).bind(wf_STATE.ITEM.PRE_FINISHED, callback);
105 	}
106 	
107 	/**
108 	 * Add a callback that will be executed on finish but after the other callbacks  
109 	 * 
110 	 * @function
111 	 * @namespace wfApi
112 	 * @param {function} callback
113 	 */
114 	function afterFinish(callback){
115 		$(window).bind(wf_STATE.ITEM.POST_FINISHED, callback);
116 	}
117 
118 }
119 
120 /*
121  * By default, the variables are pushed when the item is finished
122  */
123 afterFinish(forward);

130 /**
131  * instantiate the RecoveryContext
132  * 
133  * @function
134  * @namespace wfApi
135  * @type RecoveryContext
136  */
137 var recoveryCtx = new RecoveryContext();
138 
139 /**
140  * Initialize the interfaces communication for the context recovery.
141  * The source service defines where and how we retrieve contexts.
142  * The destination service defines where and how we send/save the contexts.
143  * 
144  * @function
145  * @namespace wfApi
146  * 
147  * @param {Object} source 
148  * @param {String} [source.type = "sync"] the type of source <b>(sync|async|manual)</b>
149  * @param {Object} [source.data] For the <i>manual</i> source type, set direclty the context data structure
150  * @param {String} [source.url = "/wfEngine/RecoveryContext/retrieve"] For the <i>sync</i> and <i>async</i> source type, the URL of the remote service
151  * @param {Object} [source.params] the parameters to send to the remote service
152  * @param {String} [source.format = "json"] the data format. <i>Only json is supported in the current version</i> 
153  * @param {String} [source.method = "post"] HTTP method of the sync service <b>(get|post)</b>
154  * 
155  * @param {Object} destination
156  * @param {String} [destination.type = "async"] the type of source <b>(sync|async)</b>
157  * @param {String} [destination.url = "/wfEngine/RecoveryContext/save"] the URL of the remote service
158  * @param {Object} [destination.params] the common parameters to send to the service
159  * @param {String} [destination.format = "json"] the data format. <i>Only json is supported in the current version</i> 
160  * @param {String} [destination.method = "post"] HTTP method of the service <b>(get|post)</b>
161  */
162 function initRecoveryContext(source, destination){
163 	recoveryCtx.initSourceService(source);
164 	recoveryCtx.initDestinationService(destination);
165 }
166 
167 /**
168  * Retrieve the context identified by the key 
169  * 
170  * @function
171  * @namespace wfApi
172  * 
173  * @param {String} key
174  * @return {Object}
175  */
176 function getRecoveryContext(key){
177 	return recoveryCtx.getContext(key);
178 }
179 
180 /**
181  * Set, send and save a context that could be recovered 
182  * 
183  * @function
184  * @namespace wfApi
185  * 
186  * @param {String} key
187  * @param {Object} data
188  */
189 function setRecoveryContext(key, data){
190 	recoveryCtx.setContext(key, data);
191 	recoveryCtx.saveContext();
192 }
193 
194 /**
195  * Remove a context (once you don't need you to recover it anymore) 
196  * 
197  * @function
198  * @namespace wfApi
199  * 
200  * @param {String} key
201  */
202 function deleteRecoveryContext(key){
203 	setRecoveryContext(key, null);
204 }
