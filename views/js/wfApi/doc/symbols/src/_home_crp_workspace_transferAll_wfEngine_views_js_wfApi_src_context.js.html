 12 /**
 13  *  The RecoveryContext enables you to initialize, 
 14  *  send and retrieve a data structure (a context).
 15  *  It can be used to recover a context in case of crash.
 16  *  
 17  * @class RecoveryContext
 18  */
 19 function RecoveryContext (){
 20 	
 21 	//keep the current instance
 22 	var _recoveryCtx = this;
 23 	
 24 	/**
 25 	 * The registry store the contexts 
 26 	 * @fieldOf RecoveryContext
 27 	 * @type {Object}
 28 	 */
 29 	this.registry = null;
 30 	
 31 	/**
 32 	 * @fieldOf RecoveryContext
 33 	 * @type {bool}
 34 	 */
 35 	this.enabled = true;
 36 	
 37 	/**
 38 	 * The parameters defining how and where to retrieve a context
 39 	 * @fieldOf RecoveryContext
 40 	 * @type {Object}
 41 	 */
 42 	this.sourceService = {
 43 			type:	'sync',										// (async | sync | manual)
 44 			data:	null,										//if type is manual, contains the data in JSON, else it should be null
 45 			url:	root_url + '/wfEngine/RecoveryContext/retrieve',		//the url where we retrieve the context
 46 			params: {},	 										//the common parameters to send to the service
 47 			method: 'post',										//sending method
 48 			format: 'json'										//the response format, now ONLY JSON is supported
 49 	};
 50 	
 51 	/**
 52 	 * The parameters defining how and where to send a context
 53 	 * @fieldOf RecoveryContext
 54 	 * @type {Object}
 55 	 */
 56 	this.destinationService = {
 57 			type:	'sync',									// (async | sync)
 58 			url:	root_url + '/wfEngine/RecoveryContext/save',			//the url where we send the context
 59 			params:  {},										//the common parameters to send to the service
 60 			method: 'post',										//sending method
 61 			format: 'json',										//the response format, now ONLY JSON is supported
 62 			flush:  true										//clear the context registry once the context is saved
 63 	};
 64 	
 65 	/**
 66 	 * Initialize the service interface for the source service: 
 67 	 * how and where we retrieve a context
 68 	 * @methodOf RecoveryContext
 69 	 * @param {Object} environment
 70 	 */
 71 	this.initSourceService = function(environment){
 72 		
 73 		//define the source service
 74 		if($.isPlainObject(environment)){
 75 			
 76 			if(environment.type){
 77 				if($.inArray(environment.type, ['manual','sync', 'async']) > -1){
 78 					this.sourceService.type = environment.type;
 79 				}
 80 			}
 81 			//manual behaviour
 82 			if(this.sourceService.type == 'manual' && $.isPlainObject(environment.data)){
 83 				this.sourceService.data = environment.data;
 84 			}
 85 			else{ 	//remote behaviour
 86 		
 87 				if(environment.url){
 88 					if(/(\/[-a-z\d%_.~+]*)*(\?[;&a-z\d%_.~+=-]*)?(\#[-a-z\d_]*)?$/.test(environment.url)){	//test url
 89 						this.sourceService.url = environment.url;		//set url
 90 					}
 91 				}
 92 				//ADD parameters
 93 				if($.isPlainObject(environment.params)){	
 94 					for(key in environment.params){
 95 						if($.inArray((typeof environment.params[key]).toLowerCase(), ['string', 'number', 'int', 'float', 'boolean']) > -1){
 96 							this.sourceService.params[key] = environment.params[key]; 
 97 						}
 98 					}
 99 				}
100 				
101 				if(environment.method){
102 					if(/^get|post$/i.test(environment.method)){
103 						this.sourceService.method = environment.method;
104 					}
105 				}
106 			}
107 		}
108 	};
109 	
110 	/**
111 	 * Retrieve a context and populate the registry
112 	 * @methodOf RecoveryContext
113 	 */
114 	this.retrieveContext = function(){
115 			
116 		if(this.sourceService.type == 'manual'){
117 			this.registry = this.sourceService.data;
118 		}
119 		else{
120 			var ctxResponse = $.ajax({
121 				async		: false,
122 				url			: this.sourceService.url,
123 				data		: this.sourceService.params,
124 				type		: this.sourceService.method,
125 				dataType	: this.sourceService.format
126 			}).responseText;
127 			try{
128 				this.registry = $.parseJSON(ctxResponse);
129 			}
130 			catch(jsonException){ console.log(ctxResponse); }
131 			
132 		}
133 	};
134 	
135 	/**
136 	 * Initialize the service interface forthe destination service: 
137 	 * how and where we send the contexts
138 	 * @methodOf RecoveryContext
139 	 * @param {Object} environment
140 	 */
141 	this.initDestinationService = function(environment){
142 		
143 		if($.isPlainObject(environment)){
144 			
145 			if(environment.type){
146 				if($.inArray(environment.type, ['sync', 'async']) > -1){
147 					this.destinationService.type = environment.type;
148 				}
149 			}
150 			if(environment.url){
151 				if(/(\/[-a-z\d%_.~+]*)*(\?[;&a-z\d%_.~+=-]*)?(\#[-a-z\d_]*)?$/.test(environment.url)){	//test url
152 					this.destinationService.url = environment.url;		//set url
153 				}
154 			}
155 			//ADD parameters
156 			if($.isPlainObject(environment.params)){	
157 				for(key in environment.params){
158 					if($.inArray((typeof environment.params[key]).toLowerCase(), ['string', 'number', 'int', 'float', 'boolean']) > -1){
159 						this.destinationService.params[key] = environment.params[key]; 
160 					}
161 				}
162 			}
163 			if(environment.method){
164 				if(/^get|post$/i.test(environment.method)){
165 					this.destinationService.method = environment.method;
166 				}
167 			}
168 			if(environment.flush){
169 				this.destinationService.flush = (environment.flush === true);
170 			}
171 		}
172 	};
173 	
174 	/**
175 	 * Save the contexts by sending them to the destination service 
176 	 * @methodOf RecoveryContext
177 	 */
178 	this.saveContext = function(){
179 		
180 		var registryParams = this.destinationService.params;
181 		registryParams['context'] = new Object();
182 		for(key in this.registry){
183 			registryParams['context'][key] = this.registry[key];
184 		}
185 		
186 		$.ajax({
187 				async		: (this.destinationService.type == 'async'),
188 				url  		: this.destinationService.url,
189 				data 		: registryParams,
190 				type 		: this.destinationService.method,
191 				dataType	: this.destinationService.format,
192 				success		: function(data){
193 			 		if(data.saved){
194 			 			if(_recoveryCtx.destinationService.flush){
195 			 				_recoveryCtx.registry = new Object();	//clear it but don't set it to null, to prevent retrieving
196 			 			}
197 			 		}
198 		 		}
199 			});
200 	};
201 	
202 	/**
203 	 * Get a context defined by the key. 
204 	 * If not loaded, we retrieve itâ‹…
205 	 * @methodOf RecoveryContext
206 	 * @param {String} key
207 	 * @returns {Object} the context
208 	 */
209 	this.getContext = function(key){
210 		if(this.enabled){
211 			if(this.registry == null){
212 				this.retrieveContext();
213 			}
214 			if(this.registry != null){
215 				return (this.registry[key]) ? this.registry[key] : {};
216 			}
217 		}
218 		return  {};
219 	};
220 	
221 	/**
222 	 * Create/edit a context
223 	 * @methodOf RecoveryContext
224 	 * @param {String} key
225 	 * @param {Object} value
226 	 */
227 	this.setContext = function(key, value){
228 		if(this.enabled){
229 			if(this.registry == null){
230 				this.registry = new Object();
231 			}
232 			if(key != ''){
233 				this.registry[key] = value;
234 			}
235 		}
236 	};
237 }
